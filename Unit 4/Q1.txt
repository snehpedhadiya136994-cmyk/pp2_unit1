#include <stdio.h>

int main() {
    FILE *fp;
    char ch, text[100];

    // Writing to file
    fp = fopen("test.txt", "w");
    printf("Enter text to write in file: ");
    gets(text);
    fprintf(fp, "%s", text);
    fclose(fp);

    // Reading from file
    fp = fopen("test.txt", "r");
    printf("\nContent of file:\n");
    while ((ch = fgetc(fp)) != EOF) {
        printf("%c", ch);
    }
    fclose(fp);
    return 0;
}
