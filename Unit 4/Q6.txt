#include <stdio.h>

int main() {
    FILE *fmain, *fodd, *feven;
    int num;

    fmain = fopen("DATA.txt", "w");
    for (int i = 1; i <= 10; i++) fprintf(fmain, "%d ", i);
    fclose(fmain);

    fmain = fopen("DATA.txt", "r");
    fodd = fopen("ODD.txt", "w");
    feven = fopen("EVEN.txt", "w");

    while (fscanf(fmain, "%d", &num) != EOF) {
        if (num % 2 == 0) fprintf(feven, "%d ", num);
        else fprintf(fodd, "%d ", num);
    }

    fclose(fmain); fclose(fodd); fclose(feven);
    return 0;
}
