#include <stdio.h>
#include <ctype.h>

int main() {
    FILE *fp = fopen("test.txt", "r");
    char ch;
    int u=0, l=0, s=0, d=0, v=0;

    while ((ch = fgetc(fp)) != EOF) {
        if (isupper(ch)) u++;
        if (islower(ch)) l++;
        if (isdigit(ch)) d++;
        char low = tolower(ch);
        if (low=='a'||low=='e'||low=='i'||low=='o'||low=='u') v++;
        if (!isalnum(ch)) s++;
    }
    printf("Upper: %d, Lower: %d, Digits: %d, Vowels: %d, Special: %d\n", u, l, d, v, s);
    fclose(fp);
    return 0;
}
