#include <stdio.h>

int main() {
    FILE *f1, *f2;
    char ch;

    f1 = fopen("test.txt", "r");
    f2 = fopen("temp.txt", "w");

    while ((ch = fgetc(f1)) != EOF) {
        if (ch == 'a') fputc('x', f2);
        else fputc(ch, f2);
    }
    fclose(f1); fclose(f2);
    remove("test.txt");
    rename("temp.txt", "test.txt");
    return 0;
}
