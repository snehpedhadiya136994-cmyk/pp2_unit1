#include <stdio.h>

struct Employee {
    int code;
    char name[30];
    float salary;
};

int main() {
    FILE *fp;
    struct Employee e, maxE;
    int n;

    fp = fopen("emp.txt", "w");
    printf("How many records? "); scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        printf("Enter Code, Name, Salary: ");
        scanf("%d %s %f", &e.code, e.name, &e.salary);
        fprintf(fp, "%d %s %f\n", e.code, e.name, e.salary);
    }
    fclose(fp);

    fp = fopen("emp.txt", "r");
    float maxSal = 0;
    while (fscanf(fp, "%d %s %f", &e.code, e.name, &e.salary) != EOF) {
        if (e.salary > maxSal) {
            maxSal = e.salary;
            maxE = e;
        }
    }
    printf("\nHighest Salary: %s (%.2f)\n", maxE.name, maxE.salary);
    fclose(fp);
    return 0;
}
