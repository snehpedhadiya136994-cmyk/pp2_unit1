#include <stdio.h>

int main() {
    FILE *fs, *ft;
    char ch;

    fs = fopen("source.txt", "r");
    ft = fopen("target.txt", "w");

    if (fs == NULL || ft == NULL) return 1;

    while ((ch = fgetc(fs)) != EOF) {
        fputc(ch, ft);
    }

    printf("File copied successfully.");
    fclose(fs);
    fclose(ft);
    return 0;
}
