
#include <stdio.h>

struct student {
    int rollno;
    char name[20];
    float score;
};

int main() {
    struct student s[5];
    int i;

    for(i=0;i<5;i++) {
        scanf("%d %s %f",
              &s[i].rollno,
              s[i].name,
              &s[i].score);
    }

    printf("\nStudents with score > 5:\n");
    for(i=0;i<5;i++) {
        if(s[i].score > 5.0) {
            printf("%s\n", s[i].name);
        }
    }

    return 0;
}
