
#include <stdio.h>

struct date {
    int d, m, y;
};

struct student {
    int roll;
    char name[20];
    struct date doj;
};

int main() {
    struct student s;

    scanf("%d %s %d %d %d",
          &s.roll,
          s.name,
          &s.doj.d,
          &s.doj.m,
          &s.doj.y);

    printf("Roll: %d\n", s.roll);
    printf("Name: %s\n", s.name);
    printf("DOJ: %d/%d/%d\n",
           s.doj.d, s.doj.m, s.doj.y);

    return 0;
}
