
#include <stdio.h>

struct date {
    int d, m, y;
};

int main() {
    struct date d1, d2;

    scanf("%d %d %d", &d1.d, &d1.m, &d1.y);
    scanf("%d %d %d", &d2.d, &d2.m, &d2.y);

    if(d1.d==d2.d && d1.m==d2.m && d1.y==d2.y)
        printf("equal");
    else
        printf("not equal");

    return 0;
}
