#include <stdio.h>

int binarySearch(int a[], int low, int high, int key) {
    if (low <= high) {
        int mid = (low + high) / 2;
        if (a[mid] == key) return mid;
        if (a[mid] > key) return binarySearch(a, low, mid - 1, key);
        return binarySearch(a, mid + 1, high, key);
    }
    return -1;
}

int main() {
    int a[10], n, key, result;
    printf("Enter N (Sorted): "); scanf("%d", &n);
    for(int i=0; i<n; i++) scanf("%d", &a[i]);
    printf("Search for: "); scanf("%d", &key);

    result = binarySearch(a, 0, n - 1, key);
    if(result != -1) printf("Found at index %d", result);
    else printf("Not found");
    return 0;
}
