#include <stdio.h>

int findMax(int a, int b) { return (a > b) ? a : b; }
int findMin(int a, int b) { return (a < b) ? a : b; }

int main() {
    int x, y;
    int (*ptr)(int, int); // Function pointer

    printf("Enter two numbers: ");
    scanf("%d %d", &x, &y);

    ptr = findMax;
    printf("Maximum: %d\n", ptr(x, y));

    ptr = findMin;
    printf("Minimum: %d\n", ptr(x, y));

    return 0;
}
