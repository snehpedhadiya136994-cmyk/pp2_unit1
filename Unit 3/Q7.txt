#include <stdio.h>

int main() {
    int r = 2, c = 2;
    int m1[2][2] = {{1, 2}, {3, 4}}, m2[2][2] = {{5, 6}, {7, 8}}, sum[2][2];
    int *p1[2], *p2[2], *pS[2];

    for(int i=0; i<2; i++) {
        p1[i] = m1[i]; p2[i] = m2[i]; pS[i] = sum[i];
    }

    for(int i=0; i<r; i++) {
        for(int j=0; j<c; j++) {
            *(pS[i] + j) = *(p1[i] + j) + *(p2[i] + j);
            printf("%d ", sum[i][j]);
        }
        printf("\n");
    }
    return 0;
}
