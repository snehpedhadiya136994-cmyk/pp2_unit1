#include <stdio.h>
#include <string.h>

int main() {
    char s[20];
    int flag = 0;
    printf("Enter string: "); scanf("%s", s);
    int len = strlen(s);

    for(int i=0; i<len/2; i++) {
        if(s[i] != s[len-1-i]) {
            flag = 1;
            break;
        }
    }

    if(flag) printf("Not a Palindrome\n");
    else printf("Palindrome\n");
    return 0;
}
