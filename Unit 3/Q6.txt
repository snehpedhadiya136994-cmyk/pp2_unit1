#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, *arr, evenSum = 0, oddSum = 0;
    printf("Enter N: ");
    scanf("%d", &n);
    arr = (int*)malloc(n * sizeof(int));

    for(int i=0; i<n; i++) {
        scanf("%d", &arr[i]);
        if(arr[i] % 2 == 0) evenSum += arr[i];
        else oddSum += arr[i];
    }

    printf("Even Sum: %d, Odd Sum: %d\n", evenSum, oddSum);
    free(arr);
    return 0;
}
